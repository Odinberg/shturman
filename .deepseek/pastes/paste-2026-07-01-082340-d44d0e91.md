from sqlalchemy import select

from sqlalchemy.exc import IntegrityError

from sqlalchemy.ext.asyncio import AsyncSession

from app.models.models import User

async def get_or_create_user(db: AsyncSession, vk_user_id: str) -> User:

"""

Атомарно находит или создаёт пользователя по vk_user_id.

Защита от race condition через retry при IntegrityError.

"""

for attempt in range(3):

result = await db.execute(

select(User).where(User.vk_user_id == vk_user_id)

)

user = result.scalar_one_or_none()

if user:

return user

try:

user = User(username=f"vk_{vk_user_id}", vk_user_id=vk_user_id)

db.add(user)

await db.flush()

return user

except IntegrityError:

await db.rollback()

if attempt == 2:

raise

continue  # retry SELECT

raise RuntimeError("Failed to get_or_create_user after 3 attempts")
